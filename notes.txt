App
  this.pics = picList;
  this.imageComponents = [img1, img2]
  this.startingImage = startingImage

  this.state: {
    img1: {
      opacity: new Animated.Value(0),
      source: null,
    },
    img2: {
      opacity: new Animated.Value(1),
      source: this.startingImage,
    },
    imgRefs = [img1, img2],
  }

  handleClick(){

    this.setState({ imgRefs: this.state.imgRefs.reverse() }, this.updateImages)

  }

  updateImages(){

    // update the images if they have run out
    if(this.pics.length === 0){
      this.pics = picList
    }

    const newState = {...this.state};
    const ref1 = imgRefs[0]
    newState[ref1]['source'] = this.pics.shift();
    this.setState(newState, this.fadeImages);

  }

  fadeImages(){

    const ref1 = imgRefs[0]
    const ref2 = imgRefs[1]

    Animated.parallel([
      Animated.timing(this.state[ref1][opacity], {
        toValue: 1,
        duration: 1500,
      }),
      Animated.timing(this.state[ref2][opacity], {
        toValue: 0,
        duration: 1500,
      }),
    ]).start();

  }

  render(){

    return(

      TouchableHighlight onPress={this.handleClick}

        <Image
          source={this.state.img1.source}
          style={[
            styles.image,
            { opacity: this.img1.opacity}
          ]}
        />

        <Image
          source={this.state.img2.source}
          style={[
            styles.image,
            { opacity: this.img2.opacity}
          ]}
        />


    )

  }
